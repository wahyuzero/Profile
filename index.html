Lightweight Fetch Wrapper

// Tiny fetch wrapper (just 12 lines!)
const api = {
  get: async (url) => {
    try {
      const res = await fetch(url);
      return await res.json();
    } catch (err) {
      console.error('API Error:', err);
      return null;
    }
  }
};
